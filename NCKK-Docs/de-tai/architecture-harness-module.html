6 Ky thuat chinh: Dependency Injection giai thich

Strategy Pattern — Moi tang co nhieu cach thuc hien

# Interface (hop dong)
class ContextStrategy(ABC):
    def build(self, history) -> str:
        ...  # moi class con tu implement

# Implementation 1: Giu toan bo
class FullHistoryStrategy(ContextStrategy):
    def build(self, history):
        return "\n".join(history)  # giu het

# Implementation 2: Chi giu 10 buoc
class WindowStrategy(ContextStrategy):
    def build(self, history):
        return "\n".join(history[-10:])  # 10 buoc cuoi

# Implementation 3: Nen thich ung
class ACCStrategy(ContextStrategy):
    def build(self, history):
        return self.compress(history)  # nen thong minh

Factory Pattern — Chon implementation tu config

# Doc config YAML → tao dung object
def create_experiment(config):
    tool = ToolFactory.create(config["tool_provider"])
    ctx  = ContextFactory.create(config["context_strategy"])
    llm  = LLMFactory.create(config["llm_backend"])
    return ModularHarness(tool, ctx, llm)

# Su dung
harness = create_experiment({
    "tool_provider": "minimal",
    "context_strategy": "acc",
    "llm_backend": "claude-sonnet"
})
result = harness.solve(task)  # chay!
